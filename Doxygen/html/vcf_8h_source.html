68 /* === Dictionary ===
69 
70  The header keeps three dictonaries. The first keeps IDs in the
71  "FILTER/INFO/FORMAT" lines, the second keeps the sequence names and lengths
72  in the "contig" lines and the last keeps the sample names. bcf_hdr_t::dict[]
73  is the actual hash table, which is opaque to the end users. In the hash
74  table, the key is the ID or sample name as a C string and the value is a
75  bcf_idinfo_t struct. bcf_hdr_t::id[] points to key-value pairs in the hash
76  table in the order that they appear in the VCF header. bcf_hdr_t::n[] is the
77  size of the hash table or, equivalently, the length of the id[] arrays.
78 */

194 /*
195  The bcf1_t structure corresponds to one VCF/BCF line. Reading from VCF file
196  is slower because the string is first to be parsed, packed into BCF line
197  (done in vcf_parse), then unpacked into internal bcf1_t structure. If it
198  is known in advance that some of the fields will not be required (notably
199  the sample columns), parsing of these can be skipped by setting max_unpack
200  appropriately.
201  Similarly, it is fast to output a BCF line because the columns (kept in
202  shared.s, indiv.s, etc.) are written directly by bcf_write, whereas a VCF
203  line must be formatted in vcf_format.
204  */
205 typedef struct {
206  int32_t rid; // CHROM
207  int32_t pos; // POS
208  int32_t rlen; // length of REF
209  float qual; // QUAL
210  uint32_t n_info:16, n_allele:16;
211  uint32_t n_fmt:8, n_sample:24;
212  kstring_t shared, indiv;
213  bcf_dec_t d; // lazy evaluation: $d is not generated by bcf_read(), but by explicitly calling bcf_unpack()
214  int max_unpack; // Set to BCF_UN_STR, BCF_UN_FLT, or BCF_UN_INFO to boost performance of vcf_parse when some of the fields won't be needed
215  int unpacked; // remember what has been unpacked to allow calling bcf_unpack() repeatedly without redoing the work
216  int unpack_size[3]; // the original block size of ID, REF+ALT and FILTER
217  int errcode; // one of BCF_ERR_* codes
218 } bcf1_t;

765 /*
766  Note that in contrast with BCFv2.1 specification, HTSlib implementation
767  allows missing values in vectors. For integer types, the values 0x80,
768  0x8000, 0x80000000 are interpreted as missing values and 0x81, 0x8001,
769  0x80000001 as end-of-vector indicators. Similarly for floats, the value of
770  0x7F800001 is interpreted as a missing value and 0x7F800002 as an
771  end-of-vector indicator.
772  Note that the end-of-vector byte is not part of the vector.
773 
774  This trial BCF version (v2.2) is compatible with the VCF specification and
775  enables to handle correctly vectors with different ploidy in presence of
776  missing values.
777  */

809 static inline void bcf_format_gt(bcf_fmt_t *fmt, int isample, kstring_t *str)
810 {
811  #define BRANCH(type_t, missing, vector_end) { \
812  type_t *ptr = (type_t*) (fmt->p + isample*fmt->size); \
813  int i; \
814  for (i=0; i<fmt->n && ptr[i]!=vector_end; i++) \
815  { \

817  if ( !(ptr[i]>>1) ) kputc('.', str); \
818  else kputw((ptr[i]>>1) - 1, str); \
819  } \
820  if (i == 0) kputc('.', str); \
821  }
822  switch (fmt->type) {

828  }
