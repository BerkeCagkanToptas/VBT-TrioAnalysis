580 static inline int hts_reg2bin(int64_t beg, int64_t end, int min_shift, int n_lvls)
581 {
582  int l, s = min_shift, t = ((1<<((n_lvls<<1) + n_lvls)) - 1) / 7;
583  for (--end, l = n_lvls; l > 0; --l, s += 3, t -= 1<<((l<<1)+l))
584  if (beg>>s == end>>s) return t + (beg>>s);
585  return 0;
586 }

588 static inline int hts_bin_bot(int bin, int n_lvls)
589 {
590  int l, b;
591  for (l = 0, b = bin; b; ++l, b = hts_bin_parent(b)); // compute the level of bin
592  return (bin - hts_bin_first(l)) << (n_lvls - l) * 3;
593 }
