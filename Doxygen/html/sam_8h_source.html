85 /* bam_cigar_type returns a bit flag with:
86  * bit 1 set if the cigar operation consumes the query
87  * bit 2 set if the cigar operation consumes the reference
88  *
89  * For reference, the unobfuscated truth table for this function is:
90  * BAM_CIGAR_TYPE QUERY REFERENCE
91  * --------------------------------
92  * BAM_CMATCH 1 1
93  * BAM_CINS 1 0
94  * BAM_CDEL 0 1
95  * BAM_CREF_SKIP 0 1
96  * BAM_CSOFT_CLIP 1 0
97  * BAM_CHARD_CLIP 0 0
98  * BAM_CPAD 0 0
99  * BAM_CEQUAL 1 1
100  * BAM_CDIFF 1 1
101  * BAM_CBACK 0 0
102  * --------------------------------
103  */
104 #define bam_cigar_type(o) (BAM_CIGAR_TYPE>>((o)<<1)&3) // bit 1: consume query; bit 2: consume reference
